Buat database dengan nama akademik

CREATE database akademik;

Buat tabel dengan nama mahasiswa di dalam database akademik
Use akademik;
Crate Table mahasiswa ('nim' varchar(5),'nama' varchar(50),'jkel' varchar(1),'alamat' text,'tgllhr' date,Primary Key ('nim'));
